def calculate_lagrange(xs, ys):
    n = len(xs)
    C = []
    for k in range(0, n):
        c_k = 1
        for i in range(0, n):
            if i == k:
                continue
            c_k *= (xs[k] - xs[i])
        C.append( ys[k] / c_k )
        
    return C

def evaluate_lagrange(C, xs, ys, x, eps=1e-4):
    # d = np.abs(xs - x)
    # di, dm = np.argmin(d), np.min(d)
    # if dm < eps:
    #     return ys[di]
    
    n = len(C)
    y = 0
    for k in range(0,n):
        dx = 1
        for i in range(0, n):
            if i==k:
                continue
            dx *= (x - xs[i])
        y += C[k] * dx
    
    return y

def divdif(xs):
    if xs.shape[0] == 1:
        return f(xs)[0]
    
    if xs.shape[0] == 2:
        return (f(xs[1]) - f(xs[0])) / (xs[1] - xs[0])
    
    return (divdif(xs[1:]) - divdif(xs[:-1])) / (xs[-1] - xs[0])

def calculate_newton(x1, y1):
    n = len(x1)
    C = []
    
    for k in range(1,n+1):
        C.append(divdif(x1[:k]))

def evaluate_newton(C, x1, y1, x, eps=1e-4):

# diag - main diagonal
# l_diag - lower subdiagonal
# u_diag - upper subdiagonal
def prohonka(diag, l_diag, u_diag, b_):
    n = b_.shape[0]
    
    a = l_diag.copy()
    b = diag.copy()
    c = u_diag.copy()
    d = b_.copy()
    
    for i in range(1, n):
        w = a[i-1] / b[i-1]
        b[i] = b[i] - w * c[i-1]
        d[i] = d[i] - w * d[i-1]
    
    x = np.zeros(n)
    x[-1] = d[-1] / b[-1]
    for i in range(n-2, -1, -1):
        x[i] = ( d[i] - c[i] * x[i+1] ) / b[i]
        
    return x

def eval_I(n, a, b, c, d, x1, y1, x):
    I = n
    for i in range(n):
        if x < x1[i]:
            I = i-1
            break
    return max(min(I, n-1), 0)

def evaluate_splines(n, a, b, c, d, x1, y1, x):
    I = eval_I(n, a, b, c, d, x1, y1, x)
    
    dx = x - x1[I]
    return a[I] + dx * (b[I] + dx * (c[I] + dx * d[I]) ) if I is not None else None
